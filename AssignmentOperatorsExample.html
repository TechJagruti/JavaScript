let x = 10; console.log("Initial Value:"); console.log("x = " + x); x += 5; // x = x + 5 console.log("After x += 5 → " + x); x -= 3; // x = x - 3 console.log("After x -= 3 → " + x); x *= 2; // x = x * 2 console.log("After x *= 2 → " + x); x /= 4; // x = x / 4 console.log("After x /= 4 → " + x); x %= 3; // x = x % 3 console.log("After x %= 3 → " + x); x **= 2; // x = x ** 2 console.log("After x **= 2 → " + x);
